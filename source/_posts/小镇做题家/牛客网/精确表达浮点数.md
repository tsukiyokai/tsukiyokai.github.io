## 题目

将给定属于 0~1 区间的有限小数或者无限循环小数（循环节用括号标记）以最简真分数表示。

## 输入输出

输入: 0.25
输出: 1/4

输入: 0.33 (33)
输出: 1/3

输入: 0.285714 (285714)
输出: 2/7

## 代码

```c++
#include <iostream>
#include <string>
#include <cmath>
using namespace std;

long long gcd (long long x, long long y) {
    if (x < y) return gcd (y, x);
    return y == 0 ? x : gcd (y, x - y);
}

void RepresentExactly (string str) {
    string strLimited, strUnLimited;
    int nLenLimited = 0, nLenUnLimited = 0;
    long long Molecule = 0, Denominator = 0;
    int nLimited = 0, nUnLimited = 0;
    long long gcd0 = 0;
    auto Start = str.find ('(');
    if (Start != str.npos) {
        auto End = str.find (')', Start);

        strLimited = str.substr (2, Start - 2);
        strUnLimited = str.substr (Start + 1, End - Start - 1);

        nLenLimited = strLimited.size ();
        nLenUnLimited = strUnLimited.size ();

        nLimited = atoi (strLimited.c_str ());
        nUnLimited = atoi (strUnLimited.c_str ());

        Molecule = nLimited * (pow (10, nLenUnLimited) - 1) + nUnLimited;
        Denominator = pow (10, nLenLimited) * (pow (10, nLenUnLimited) - 1);
        gcd0 = gcd (Molecule, Denominator);
        cout << Molecule /gcd0 << "/" << Denominator /gcd0 << endl;
    }
    else {
        strLimited = str.substr (2);
        nLenLimited = strLimited.size ();
        Molecule = atoi (strLimited.c_str ());
        Denominator = (long long)(pow (10, nLenLimited));
        gcd0 = gcd (Molecule, Denominator);
        cout << Molecule /gcd0 << "/" << Denominator /gcd0 << endl;
    }
}

int main () {
    string str = "";
    while (cin >> str)
        RepresentExactly (str);
    return 0;
}
```
