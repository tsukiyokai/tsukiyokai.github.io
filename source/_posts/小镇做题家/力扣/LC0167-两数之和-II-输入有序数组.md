## 题目

给定一个已按照升序排列的整数数组numbers，请你从数组中找出两个数满足相加之和等于目标数target。

函数应该以长度为2的整数数组的形式返回这两个数的下标值。numbers的下标从1开始计数，所以答案数组应当满足`1<=answer[0]<answer[1]<=numbers.length`。你可以假设每个输入只对应唯一的答案，而且你不可以重复使用相同的元素。

## 解法1：二分搜索

注意对二分搜索的左右边界进行迭代。

**代码**

```c++
class Solution {
public:
    vector<int> twoSum(vector<int>& numbers, int target) {
        vector<int> res;
        for (int i = 0; i < numbers.size(); i++) {
            int find = binSearch(numbers, i + 1, numbers.size() - 1, target - numbers[i]);
            if (find == -1) continue;
            if (find != -1) {
                res.push_back(i + 1);
                res.push_back(find + 1);
            }
            break;
        }
        return res;
    }

    int binSearch(vector<int>& nums, int l, int r, int x) {
        while (l <= r) {
            int mid = l + (r - l) / 2;
            if (nums[mid] == x) return mid;
            else if (nums[mid] < x) l = mid + 1;
            else { r = mid - 1; }
        }
        return -1;
    }
};
```
