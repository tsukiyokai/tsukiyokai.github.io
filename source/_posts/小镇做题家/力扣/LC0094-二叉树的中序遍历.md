## 解法1：递归

**代码**

```c++
class Solution {
public:
    vector<int> inorderTraversal(TreeNode* root) {
        vector<int> res;
        fun(root, res);
        return res;
    }

    void fun(TreeNode* root, vector<int>& a) {
        if (root == NULL) return;
        fun(root->left, a);
        a.push_back(root->val);
        fun(root->right, a);
    }
};
```

## 解法2：迭代

**代码**

```c++
class Solution {
public:
    vector<int> inorderTraversal(TreeNode* root) {
        vector<int> res;
        fun(root, res);
        return res;
    }

    void fun(TreeNode* root, vector<int>& a) {
        stack<TreeNode*> s;
        TreeNode* curr = root;
        while (s.empty() == false || curr != nullptr) {
            while (curr != nullptr) {
                s.push(curr);
                curr = curr->left;
            }
            if (s.empty() == false) {
                curr = s.top(); s.pop();
                a.push_back(curr->val);
                curr = curr->right;
            }
        }
    }
};
```
