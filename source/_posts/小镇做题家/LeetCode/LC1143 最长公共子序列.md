给定两个字符串 X 和 Y，返回这两个字符串的最长公共子序列的长度。
如果不存在公共子序列，返回 0。

```c++
class Solution {
public:
    int longestCommonSubsequence(string X, string Y) {
        int m = X.length();
        int n = Y.length();
        vector<vector<int>> L(m + 1, vector<int>(n + 1, 0));
        for (int i = 0; i <= m; i++) {
            for (int j = 0; j <= n; j++) {
                if (i == 0 || j == 0)
                    L[i][j] = 0;
                else if (X[i - 1] == Y[j - 1])
                    L[i][j] = L[i - 1][j - 1] + 1;
                else
                    L[i][j] = max(L[i - 1][j], L[i][j - 1]);
            }
        }
        return L[m][n];
    }
};
```
