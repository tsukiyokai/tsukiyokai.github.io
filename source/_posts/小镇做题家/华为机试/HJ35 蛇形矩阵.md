蛇形矩阵是由1开始的自然数依次排列成的一个矩阵上三角形。

```c++
#include <iostream>
using namespace std;

int main() {
    int n;
    while (cin >> n) {
        int beg = 1;
        for (int i = 1; i <= n; i++) {
            cout << beg;
            int tmp = beg;
            for (int j = 2; j <= n + 1 - i; j++) {
                tmp += j + (i - 1);
                cout << ' ' << tmp;
            }
            cout << '\n';
            beg += i;
        }
    }

    return 0;
}
```

beg表示每行的开头元素
