给出一个名字，该名字有26个字符串组成，定义这个字符串的“漂亮度”是其所有字母“漂亮度”的总和。每个字母都有一个“漂亮度”，范围在1到26之间。没有任何两个字母拥有相同的“漂亮度”。字母忽略大小写。给出多个名字，计算每个名字最大可能的“漂亮度”。本题含有多组数据。

```c++
#include <bits/stdc++.h>
using namespace std;

void tolow(string& s) {
    for (char& ch : s)
        ch = tolower(ch);
}

bool cmp(pair<char, int> a1, pair<char, int> a2) {
    return a1.second > a2.second;
}

int beauty(string& s) {
    int w[26];
    for (int i = 0; i < 26; i++)
        w[i] = 26 - i;
    unordered_map<char, int> m;
    for (auto ch : s)
        m[ch] += 1;
    vector<pair<char, int>> v;
    for (auto p : m)
        v.push_back(p);
    sort(v.begin(), v.end(), &cmp);
    int res = 0;
    for (int i = 0; i < v.size(); i++)
        res += w[i] * v[i].second;

    return res;
}

int main() {
    int N;
    while (cin >> N) {
        vector<string> names;
        while (N--) {
            string tmp;
            cin >> tmp;
            tolow(tmp);
            names.push_back(tmp);
        }
        for (auto& name : names)
            cout << beauty(name) << endl;
    }
    return 0;
}
```
