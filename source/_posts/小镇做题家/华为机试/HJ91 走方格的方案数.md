## 题目

请计算 n*m 的棋盘格子（n 为横向的格子数，m 为竖向的格子数）沿着各自边缘线从左上角走到右下角，总共有多少种走法，要求不能走回头路，即：只能往右和往下走，不能往左和往上走。

## 解法 1：递归

```c++
#include <iostream>
using namespace std;

int uniquePaths(int m, int n) {
    if (m == 0 || n == 0) return 1; // 只有一条直路可以走到左或上边界的点
    return uniquePaths(m - 1, n) + uniquePaths(m, n - 1);
}

int main() {
    int m, n;
    while (cin >> n >> m) {
        cout << uniquePaths(m, n) << endl;
    }
    return 0;
}
```

## 解法 2：DP

```c++
#include <iostream>
#include <vector>
using namespace std;

/*
* (0,0)
*               (i-1,j)
*       (i,j-1) (i,  j)
*
*                           (m-1,n-1)
*/

int uniquePaths(int m, int n) { // m行 n列
    // dp[i][j]表示到该达点的走法数，注意它是路径数而非步骤数。
    vector<vector<int>> dp(m, vector<int>(n, 0));
    for (int i = 0; i < m; i++) dp[i][0] = 1;
    for (int j = 0; j < n; j++) dp[0][j] = 1;
    for (int i = 1; i < m; i++) {
        for (int j = 1; j < n; j++) {
            dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
        }
    }
    return dp[m - 1][n - 1];
}

int main() {
    int m, n;
    while (cin >> n >> m) {
        cout << uniquePaths(m + 1, n + 1) << endl;
    }
    return 0;
}
```

为什么传入是 m+1 和 n+1？因为牛客这道题走的是边缘线，而力扣上这道题走的是格子。
