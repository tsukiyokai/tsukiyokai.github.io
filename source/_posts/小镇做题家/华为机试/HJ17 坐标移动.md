开发一个坐标计算工具， A表示向左移动，D表示向右移动，W表示向上移动，S表示向下移动。从（0,0）点开始移动，从输入字符串里面读取一些坐标，并将最终输入结果输出到输出文件里面。
```
输入
A10;S20;W10;D30;X;A1A;B10A11;;A10;
输出
10,-10
```

```c++
#include<iostream>
#include<string>
#include<vector>
#include<sstream>
#include<unordered_map>
using namespace std;

bool isvalid(string s) {
    return (s[0] == 'W' || s[0] == 'A' || s[0] == 'S' || s[0] == 'D')
        && s.length() >= 2 && s.length() <= 3
        && s.substr(1).find_first_not_of("0123456789") == string::npos;
}

int main() {
    string s;
    while (getline(cin, s)) {
        vector<string> v;
        stringstream ss(s);
        string tmps;
        while (getline(ss, tmps, ';'))
            v.push_back(tmps);

        unordered_map<char, int> m;
        for (int i = 0; i < v.size(); i++) {
            if (isvalid(v[i])) {
                char a = v[i][0];
                int b = stoi(v[i].substr(1));
                m[a] += b;
            }
        }

        int x = 0 + m['D'] - m['A'];
        int y = 0 + m['W'] - m['S'];
        cout << x << "," << y << endl;
    }

    return 0;
}
```
