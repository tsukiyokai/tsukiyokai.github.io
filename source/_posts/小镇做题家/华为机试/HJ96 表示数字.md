## 题目

将一个字符中所有的整数前后加上符号'*'，其他字符保持不变。连续的数字视为一个整数。

## 解法

```c++
#include<iostream>
#include<string>
using namespace std;

int main() {
    string s;
    while (cin >> s) {
        string res;
        for (size_t i = 0; i < s.length(); i++) {
            if (!isdigit(s[i])) {
                if (i == 0 || !isdigit(s[i - 1]))
                    res += s[i];
                else {
                    res += '*';
                    res += s[i];
                }
            }
            else {
                if (i == 0 || !isdigit(s[i - 1])) {
                    res += '*';
                    res += s[i];
                }
                else {
                    res += s[i];
                }
            }
        }
        if (isdigit(res.back())) res.push_back('*');
        cout << res << endl;
    }
    return 0;
}
```
