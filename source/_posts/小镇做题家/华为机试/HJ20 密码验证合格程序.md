密码要求: 1.长度超过 8 位 2.包括大小写字母.数字.其它符号,以上四种至少三种 3.不能有相同长度大于 2 的子串重复

```c++
#include<iostream>
#include<string>
using namespace std;

inline bool rule1(string& s) { return s.length() > 8; }

bool rule2(string& s) {
    bool AA = 0, aa = 0, digit = 0, symbol = 0;
    for (char& c : s) {
        if (c >= 'A' && c <= 'Z') AA = 1;
        else if (c >= 'a' && c <= 'z') aa = 1;
        else if (c >= '0' && c <= '9') digit = 1;
        else symbol = 1;
    }
    return AA + aa + digit + symbol >= 3;
}

bool rule3(string s) {
    for (size_t len = 3; len <= s.length(); len++)
        for (size_t index = 0; index <= s.length() - len; index++)
            // if (s.find(s.substr(index, len), index + 1) != string::npos)
            if (s.substr(index + 1).find(s.substr(index, len)) != string::npos)
                return false;
    return true;
}

int main() {
    string s;
    while (getline(cin, s)) {
        if (rule1(s) && rule2(s) && rule3(s))
            cout << "OK" << endl;
        else cout << "NG" << endl;
    }

    return 0;
}
```

rule3 里面也可以写成注释的那句，用到了 find 的另一个重载：
find(string, int)：第一个参数用来指示要查找的目标串，第二个参数用来表示从字符串的何处开始查找子串(默认的查找位置是 0)。
