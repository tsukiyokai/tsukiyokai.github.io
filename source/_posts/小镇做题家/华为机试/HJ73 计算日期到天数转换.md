## 

根据输入的日期，计算是这一年的第几天。测试用例有多组，注意循环输入

## 解法

**代码**

```c++
#include<iostream>
#include<string>
#include<algorithm>
using namespace std;

bool isleap(int y) {
    if (y % 4 == 0 && y % 100 != 0 || y % 400 == 0) return true;
    return false;
}

int main() {
    int days[12] = { 31,28,31,30,31,30,31,31,30,31,30,31 };
    int y, m, d;
    while (cin >> y >> m >> d) {
        int num = 0;
        for (int i = 0; i < m - 1; i++) {
            num += days[i];
        }
        num += d;
        if (m > 2 && isleap(y)) num += 1;
        cout << num << endl;
    }

    return 0;
}
```
