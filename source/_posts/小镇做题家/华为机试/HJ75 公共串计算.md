## 题目

给定两个只包含小写字母的字符串，计算两个字符串的最大公共子串的长度。
注：子串的定义指一个字符串删掉其部分前缀和后缀（也可以不删）后形成的字符串。

## 解法：字符串

暴力法

```c++
#include <iostream>
#include <string>
using namespace std;

bool issub(string str1, string str2) { // 判断str1整体是否为str2的子串
    int len1 = str1.length();
    int len2 = str2.length();
    for (int i = 0; i <= (len2 - len1); i++) {
        if (str1 == str2.substr(i, len1)) return true;
    }
    return false;
}

int maxcommonsubstr(string s1, string s2) {
    int len0 = 0;
    int len1 = s1.length();
    int len2 = s2.length();

    for (int i = len1; i > 0; i--) {
        for (int j = 0; j <= len1 - i; j++) {
            if (issub(s1.substr(j, i), s2)) {
                len0 = i;
                break;
            }
            continue;
        }
        if (len0) break;
        else continue;
    }
    return len0;
}

int main() {
    string s1, s2;
    cin >> s1 >> s2;
    int len1 = s1.length();
    int len2 = s2.length();
    if (len1 >= len2) swap(s1, s2); // 之后按s1短s2长进行调用
    cout << maxcommonsubstr(s1, s2) << endl;
    return 0;
}
```
