# 数组 旋转数组

## 正文

将一个长度为n的数组arr[]左旋d个位置。例如初始数组为：1 2 3 4 5 6 7

例如，将上面的数组左旋2个位置得到下面的数组：3 4 5 6 7 1 2

```c++
#include <iostream>
using namespace std;

/* 绕第1个元素左旋（除第1个外其余元素依次左移，首元素arr[0]交换到最右端 */
void leftRotatebyOne(int arr[], int n) {
    int temp = arr[0], i;
    for (i = 0; i < n - 1; i++)
        arr[i] = arr[i + 1];
    arr[i] = temp;
}

/* 第d个元素的索引是arr[d-1],从arr[0]到arr[d-1]依次调用leftRotatebyOne() */
void leftRotate(int arr[], int n, int d) {
    for (int i = 0; i < d; i++)
        leftRotatebyOne(arr, n);
}

/* 打印数组 */
void printArray(int arr[], int n) {
    for (int i = 0; i < n; i++)
        cout << arr[i] << (i != n - 1 ? " " : "");
    cout << endl;
}

/* 主函数 */
int main() {
    int arr[] = { 1,2,3,4,5 };
    int n = sizeof(arr) / sizeof(arr[0]); /* n:数组长度 */
    leftRotate(arr, n, 2);
    printArray(arr, n);
    return 0;
}
```
