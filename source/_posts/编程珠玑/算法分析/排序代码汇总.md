```c++
#include <iostream>
#include <vector>
#include <algorithm>
#include <ctime>

using namespace std;

void printArray(vector<int>& arr) {
    for (auto& i : arr)
        cout << i << '\t';
    cout << endl;
}

void bubbleSort(vector<int>& arr) {
    int N = arr.size();
    bool swapped;
    for (int i = 0; i < N - 1; i++) {
        swapped = false;
        for (int j = 0; j < N - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                swap(arr[j], arr[j + 1]);
                swapped = true;
            }
        }
        if (!swapped) break;
    }
}

void insertionSort(vector<int>& arr) {
    int N = arr.size();
    for (int i = 1; i < N; i++) {
        for (int j = i - 1; j >= 0; j--) {
            if (arr[j] > arr[j + 1])
                swap(arr[j], arr[j + 1]);
            else break;
        }
    }
}

void selectionSort(vector<int>& arr) {
    int N = arr.size();
    int i, j;
    for (i = 0; i < N - 1; i++) {
        int min = i;
        for (j = i + 1; j < N; j++)
            if (arr[j] < arr[min]) min = j;
        swap(arr[min], arr[i]);
    }
}

int partition(vector<int>& arr, int left, int right) {
    int pivot = arr[right];
    while (left < right) {
        while (left < right && arr[left] < pivot) left++;
        arr[right] = arr[left];
        while (left < right && arr[right] > pivot) right--;
        arr[left] = arr[right];
    }
    arr[left] = pivot;
    return left;
}

void quickSort(vector<int>& arr, int low, int high) {
    if (low < high) {
        int p = partition(arr, low, high);
        quickSort(arr, low, p - 1);
        quickSort(arr, p + 1, high);
    }
}

void quickSort(vector<int>& arr) {
    int low = 0, high = arr.size() - 1;
    quickSort(arr, low, high);
}

int main() {
    int tstart, tstop;
    vector<int> arr(100);
    srand(static_cast<unsigned>(time(0)));
    generate(arr.begin(), arr.end(), rand);
    cout << "排序前: " << endl;
    printArray(arr);
    tstart = clock();

    quickSort(arr);

    //sort(arr.begin(), arr.end());
    tstop = clock();
    cout << "排序后: " << endl;
    printArray(arr);
    cout << "排序时间：" << endl << tstop - tstart << endl;
    /*
    * 排序1W数据耗时：
    * 冒泡 4900   ms
    * 插入 4100   ms
    * 选择 1700   ms
    * 标准   18   ms
    * 快速 递归太深，跑没了
    */

    return 0;
}
```
