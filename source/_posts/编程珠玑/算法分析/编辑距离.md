Given two strings str1 and str2 and below operations that can performed on str1. Find minimum number of edits (operations) required to convert ‘str1’ into ‘str2’.

Insert
Remove
Replace
All of the above operations are of equal cost.

递归解法

```c++
#include <bits/stdc++.h>
using namespace std;

int min(int x, int y, int z) { return min(min(x, y), z); }

int editDist(string str1, string str2) {
    int m = str1.length();
    int n = str2.length();

    if (m == 0 || n == 0) return m == 0 ? n : m;

    if (str1[m - 1] == str2[n - 1])
        return editDist(str1.substr(0, m - 1), str2.substr(0, n - 1));

    return 1 +
        min(editDist(str1, str2.substr(0, n - 1)), // Insert
            editDist(str1.substr(0, m - 1), str2), // Remove
            editDist(str1.substr(0, m - 1), str2.substr(0, n - 1)) // Replace
        );
}

int main() {
    string str1 = "lixing";
    string str2 = "fanshanshan";

    cout << editDist(str1, str2);

    return 0;
}
```
