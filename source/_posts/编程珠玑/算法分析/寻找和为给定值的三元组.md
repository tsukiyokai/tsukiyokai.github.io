Given an array and a value, find if there is a triplet in array whose sum is equal to the given value. If there is such a triplet present in array, then print the triplet and return true. Else return false.

解法1：暴力法

三层for，时间复杂度n^3

解法二：双指针

时间复杂度n^2

```c++
#include <bits/stdc++.h>
using namespace std;

bool find3Numbers(vector<int> A, int sum) {
    int N = A.size();
    sort(A.begin(), A.end());
    int left, right;
    for (int i = 0; i < N - 2; i++) {
        left = i + 1;
        right = N - 1;
        while (left < right) {
            if (A[i] + A[left] + A[right] == sum) {
                cout << "Triplet is " << A[i] << ", " << A[left] << ", " << A[right];
                return true;
            }
            else if (A[i] + A[left] + A[right] < sum) left++;
            else right--;
        }
    }
    return false;
}

int main() {
    vector<int> A = { 1, 4, 45, 6, 10, 8 };
    int sum = 22;
    find3Numbers(A, sum);
    return 0;
}
```

解法3：哈希

```c++
#include <bits/stdc++.h>
using namespace std;

bool find3Numbers(vector<int> A, int sum) {
    int arr_size = A.size();
    for (int i = 0; i < arr_size - 2; i++) {
        unordered_set<int> uset;
        int curr_sum = sum - A[i];
        for (int j = i + 1; j < arr_size; j++) {
            if (uset.find(curr_sum - A[j]) != uset.end()) {
                printf("Triplet is %d, %d, %d", A[i], A[j], curr_sum - A[j]);
                return true;
            }
            uset.insert(A[j]);
        }
    }
    return false;
}

int main() {
    vector<int> A = { 1, 4, 45, 6, 10, 8 };
    int sum = 22;
    find3Numbers(A, sum);
    return 0;
}
```
