注意一点，S2 不为空时，就不能继续让 S1 向 S2 转移，必须把 S2 中的元素 De 完再继续 S1 向 S2 转移。比如 input1 2 3，如果 S2 中 1 2 出列，3 还留在里面，此时 input4 5，整体来看一共 input 了 1 2 3 4 5，应该 output1 2 3 4 5，所以 4 5 不能压在 S2 中仅存的 3 的上面，一定要把 S2 容器 De 空，然后再继续转移。

```c++
struct Queue {
    stack<int> s1, s2;

    void enQueue(int x) {
        s1.push(x);
    }

    int deQueue() {
        if (s1.empty() && s2.empty()) {
            cout << "Q is empty";
            exit(0);
        }

        if (s2.empty()) {
            while (!s1.empty()) {
                s2.push(s1.top());
                s1.pop();
            }
        }

        int x = s2.top();
        s2.pop();
        return x;
    }
};
```

#### Implementation of method 2 using Function Call Stack

法二的另一种实现方式，使用了递归思路，也不容易理清，还是推荐第二种办法。

```c++
struct Queue {
    stack<int> s;

    void enQueue(int x) {
        s.push(x);
    }

    int deQueue() {
        if (s.empty()) {
            cout << "Q is empty";
            exit(0);
        }

        int x = s.top();
        s.pop();
        if (s.empty()) return x;
        int item = deQueue();
        s.push(x);
        return item;
    }
};
```
