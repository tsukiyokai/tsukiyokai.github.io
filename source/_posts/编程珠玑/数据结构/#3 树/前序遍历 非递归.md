# 前序遍历 非递归

## 必背

```c++
void preorderIterative(Node* root)
{
    if (root == NULL) return;

    stack<Node*> st;
    Node* curr = root;

    while (curr || !st.empty())
    {
        while (curr != NULL)
        {
            cout << curr->data << " ";

            if (curr->right)
                st.push(curr->right);

            curr = curr->left;
        }

        if (!st.empty()) {
            curr = st.top();
            st.pop();
        }
    }
}
```

## 正文

```c++
#include <bits/stdc++.h>
using namespace std;

// Tree Node
struct Node {
    int data;
    Node* left, * right;
    Node(int data)
    {
        this->data = data;
        this->left = this->right = NULL;
    }
};

// Iterative function to do Preorder traversal of the tree
void preorderIterative(Node* root)
{
    if (root == NULL) return;

    stack<Node*> st;

    // start from root node (set current node to root node)
    Node* curr = root;

    // run till stack is not empty or current is
    // not NULL
    while (curr || !st.empty())
    {
        // Print left children while exist
        // and keep pushing right into the
        // stack.
        while (curr != NULL)
        {
            cout << curr->data << " ";

            if (curr->right) // 搞清楚，这里curr并未移动，只是进行了判断和入栈的操作。要搞清楚什么呢？就是栈里存的东西究竟是什么。
                // 在inorder非递归中，当时栈里保存的是一路走过的根结点，然后再向左移动。之后这些保存在栈里的根结点需要在右子树操作前输出
                // 因此！无论in还是pre非递归用到的stack中保存的东西，就是需要在输出左结点之后的“第二时间”要输出的东西。
                // 例如inorder中，顺序为左根右，stack就保存到达左后第二时间要输出的（一路向左过程中所经过的）根结点
                // 则preorder中，顺序为根左右，stack就保存到达左后第二时间要输出的右子树结点（就像下面的st.push(curr->right);）
                // 因为紧挨着左后就要输出了，所以就需要再到达左之前就保存进栈里，到时候直接弹出即可
                st.push(curr->right);

            curr = curr->left;
        }

        // We reach when curr is NULL, so We
        // take out a right child from stack
        if (!st.empty()) {
            curr = st.top();
            st.pop();
        }
    }
}

// Driver Code
//          10
//        /    \
//       20    30
//      /  \   /
//     40  50 60
//    /  \
//   70  80
//
int main()
{
    Node* root = new Node(10);
    root->left = new Node(20);
    root->right = new Node(30);
    root->left->left = new Node(40);
    root->left->left->left = new Node(70);
    root->left->right = new Node(50);
    root->right->left = new Node(60);
    root->left->left->right = new Node(80);

    preorderIterative(root);

    return 0;
}
```
