# 二叉树 删除

给定一个二叉树，通过确保树从底部收缩（即，被删除的结点被最右下的那个结点替换）从中删除一个结点。这与二叉查找树的删除不同。这里元素之间没有任何顺序，所以我们用最后一个元素替换。

示范

```
Delete 10 in below tree
       10
     /    \         
    20     30
Output :    
       30
     /             
    20     


Delete 20 in below tree
       10
     /    \         
    20     30
            \
            40
Output :    
       10
     /   \             
    40    30
```

算法

从根结点开始，找到二叉树中最右下的结点和要删除的结点。分别用两个指针变量将其保存起来

用最右下结点的值替代待删除结点。

然后删除最深的最右边的结点。

```c++
// C++ program to delete element in binary tree
#include <bits/stdc++.h>
using namespace std;

/* A binary tree node has key, pointer to left
child and a pointer to right child */
struct Node {
    int key;
    struct Node* left, * right;
};

/* function to create a new node of tree and
return pointer */
struct Node* newNode(int key)
{
    struct Node* temp = new Node;
    temp->key = key;
    temp->left = temp->right = NULL;
    return temp;
};

/* Inorder traversal of a binary tree*/
void inorder(struct Node* temp)
{
    if (!temp)
        return;
    inorder(temp->left);
    cout << temp->key << " ";
    inorder(temp->right);
}

/* function to delete the given deepest node
(d_node) in binary tree */
void deletDeepest(struct Node* root, struct Node* d_node) // root指向根结点，d_node指向最右下结点
{
    queue<struct Node*> q;
    q.push(root);

    // Do level order traversal until last node
    struct Node* temp;
    while (!q.empty()) {
        temp = q.front(); // temp将表示d_node（最右下结点）的父结点
        q.pop();
        if (temp == d_node) {
            temp = NULL;
            delete (d_node);
            return;
        }
        if (temp->right) {
            if (temp->right == d_node) {
                temp->right = NULL;
                delete (d_node);
                return;
            }
            else
                q.push(temp->right);
        }

        if (temp->left) {
            if (temp->left == d_node) {
                temp->left = NULL;
                delete (d_node);
                return;
            }
            else
                q.push(temp->left);
        }
    }
}

/* function to delete element in binary tree */
Node* deletion(struct Node* root, int key)
{
    if (root == NULL)
        return NULL;

    if (root->left == NULL && root->right == NULL) {
        if (root->key == key)
            return NULL;
        else
            return root;
    }

    queue<struct Node*> q;
    q.push(root);

    struct Node* temp = NULL;
    struct Node* key_node = NULL;

    // Do level order traversal to find deepest
    // node(temp) and node to be deleted (key_node)
    while (!q.empty()) { // 只要队列不为空，就循环，循环出来时，temp指向为最右下的结点
        temp = q.front();
        q.pop();

        if (temp->key == key) // 如果temp是待删除的结点
            key_node = temp; // 则令keynode和temp指向同一内容，即keynode也指向temp的左右子树，同时temp的父结点也指向keynode

        if (temp->left)
            q.push(temp->left);

        if (temp->right)
            q.push(temp->right);
    }

    if (key_node != NULL) {
        int x = temp->key; // 循环刚刚结束，此时temp->key是最右下结点的数据域，把它存在x里待用
        deletDeepest(root, temp); // 删除右下结点temp
        key_node->key = x; // 待删除结点键值改成最右下结点
    }
    return root;
}

// Driver code
int main()
{
    struct Node* root = newNode(10);
    root->left = newNode(11);
    root->left->left = newNode(7);
    root->left->right = newNode(12);
    root->right = newNode(9);
    root->right->left = newNode(15);
    root->right->right = newNode(8);

    cout << "Inorder traversal before deletion : ";
    inorder(root);

    int key = 11;
    root = deletion(root, key); // 删除root所指向的树中键值为key的结点

    cout << endl;
    cout << "Inorder traversal after deletion : ";
    inorder(root);

    return 0;
}
```
