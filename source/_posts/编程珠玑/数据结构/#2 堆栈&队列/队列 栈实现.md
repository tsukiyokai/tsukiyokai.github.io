注意一点，S2不为空时，就不能继续让S1向S2转移，必须把S2中的元素De完再继续S1向S2转移。比如input1 2 3，如果S2中1 2出列，3还留在里面，此时input4 5，整体来看一共input了1 2 3 4 5，应该output1 2 3 4 5，所以4 5不能压在S2中仅存的3的上面，一定要把S2容器De空，然后再继续转移。

```c++
struct Queue {
    stack<int> s1, s2; // s2|s1

    // Enqueue an item to the queue
    void enQueue(int x) {
        // Push item into the first stack
        s1.push(x);
    }

    // Dequeue an item from the queue
    int deQueue() { /*不新传入任何参数*/
        // if both stacks are empty
        if (s1.empty() && s2.empty()) {
            cout << "Q is empty";
            exit(0);
        }

        // if s2 is empty, move
        // elements from s1
        if (s2.empty()) {
            while (!s1.empty()) {
                s2.push(s1.top());
                s1.pop();
            }
        }

        // return the top item from s2
        int x = s2.top();
        s2.pop();
        return x;
    }
};
```

## Implementation of method 2 using Function Call Stack

法二的另一种实现方式，使用了递归思路，也不容易理清，还是推荐第二种办法。

```c++
struct Queue {
    stack<int> s;

    // Enqueue an item to the queue
    void enQueue(int x) {
        s.push(x);
    }

    // Dequeue an item from the queue
    int deQueue() {
        if (s.empty()) {
            cout << "Q is empty";
            exit(0);
        }

        // pop an item from the stack
        int x = s.top();
        s.pop();

        // if stack becomes empty, return
        // the popped item
        if (s.empty()) return x;

        // recursive call
        int item = deQueue();

        // push popped item back to the stack
        s.push(x);

        // return the result of deQueue() call
        return item;
    }
};
```
